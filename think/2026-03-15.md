acceptableHallucinationRate: 0.01, // 1% 可接受幻觉

// 第1层：预防（创造者设计）

// 第2层：检测（观察者监控）

// 第3层：保护（自动响应）
ProtectionBarrier(actions: [suspend, escalate, degradeGracefully]),

];
// 3. 资本缓冲（Basel 风格）
